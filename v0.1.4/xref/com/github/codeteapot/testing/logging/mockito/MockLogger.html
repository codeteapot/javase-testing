10  /**
11   * Marks a handler as a substitute for a logger.
12   *
13   * <p>The annotated target must be of type {@link Handler}. This is assigned the result of a call to
14   * {@link Mockito#mock(Class)} and used as a parameter to the underlying call to
15   * {@link LoggerStub#stubFor(String, Handler)}.
16   */
17  @Retention(RUNTIME)
18  public @interface MockLogger {
19  
20    /**
21     * Name of the logger to replace.
22     *
23     * @return The name of the logger.
24     */
25    String name();
26  }
